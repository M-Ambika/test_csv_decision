def self.matches?(cell)
  return false unless (match = EXPRESSION.match(cell))

  proc = non_numeric?(match)
  return proc if proc

  numeric?(match)
end

def self.numeric?(match)
  value = Matchers.to_numeric(match['value'])
  return false unless value

  proc(function: value)
end
