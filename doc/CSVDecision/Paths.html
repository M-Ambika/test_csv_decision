def self.scan(table:)
  # Do we even have paths?
  columns = table.columns.paths.keys
  return [] if columns.empty?

  table.paths = Paths.new(table: table, columns: columns).paths
end

def self.value(current_value, index)

  current_value[-1] = index
  current_value
